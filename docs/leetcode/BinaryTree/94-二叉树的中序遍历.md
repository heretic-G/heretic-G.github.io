```javascript

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
function inorderTraversal (root) {
    let arr = []
    function deal (root) {
        if (!root) return
        if (root.left){
            deal(root.left)
        }
        arr.push(root.val)
        if (root.right){
            deal(root.right)
        }
    }
    deal(root)
    return arr
} 

```

前中后的差异就是根的位置 

前就是根左右

中就是左根右

后就是左右根
