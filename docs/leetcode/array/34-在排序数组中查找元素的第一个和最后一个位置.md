var searchRange = function(nums, target) {

    let left = 0
    let right = nums.length - 1
    let nextR = right
    while (left < right) {
        let mid = (left + right) >> 1
        if (nums[mid] < target) {
            left = mid + 1
        } else {
            if (nums[mid] > target) {
                nextR = mid
            }
            right = mid - 1
        }
    }
    let result = []
    if (nums[left]!== target) left += 1
    result.push(left)
    right = nextR
    while (left < right) {
        let mid = (left + right) >> 1
        if (nums[mid] > target) {
            right = mid - 1
        } else {
            left = mid + 1
        }
    }
    if (nums[right]!== target) right -= 1

};
