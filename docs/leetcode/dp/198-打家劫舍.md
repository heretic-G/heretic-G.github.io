/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function(nums) {
    if (nums.length < 3) return Math.max(...nums)
    let arr = [nums[0], Math.max(nums[1], nums[0])]
    let i = 2
    while (i < nums.length) {
        arr[i] = Math.max(arr[i - 1], arr[i - 2] + nums[i])
        i++
    }
    return Math.max(arr.pop(), arr.pop())
};
