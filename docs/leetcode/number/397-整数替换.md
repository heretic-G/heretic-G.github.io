var integerReplacement = function(n) {
    let num = 0
    while (n > 3) {
        if (n & 1 === 1) {
            if ((n & 3) === 3) {
                n += 1
            } else {
                n -= 1
            }
        } else {
            n /= 2 
        }
        num += 1
    }
    return num + n - 1
};
